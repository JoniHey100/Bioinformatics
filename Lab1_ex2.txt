#include <iostream>
#include <string>
#include <map>
#include <iomanip> 
using namespace std;

int main() {
    string seq;
    cout << "Sequence: ";
    cin >> seq;

    map<char, int> freq;

    for (char base : seq) {
        freq[base]++;
    }

    int total = seq.length();

    cout << "Alphabet: ";
    for (auto &p : freq) {
        cout << p.first << " ";
    }
    cout << endl;

    cout << "\nFrequency:" << endl;
    for (auto &p : freq) {
        double percent = (double)p.second / total * 100;
        cout << p.first << " - " << fixed << setprecision(2) << percent << "%" << endl;
    }

    return 0;
}
